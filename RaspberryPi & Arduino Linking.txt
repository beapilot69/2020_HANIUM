#라즈베리 파이 소스코드, 아두이노에서 시리얼 통신으로 보낸 값을 라즈베리 파이에서 반복해서 읽고 출력
import serial

port="/dev/ttyACM0"
serialFromArduino = serial.Serial(port, 9600)
serialFromArduino.flushInput()
while True:
    input_s = serialFromArduino.readline()
    input = int(input_s)
    print(input * 10)
